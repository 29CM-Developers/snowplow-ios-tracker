50 - (void)addOrReplaceStateMachine:(id<SPStateMachineProtocol>)stateMachine identifier:(NSString *)stateMachineIdentifier {
51  @synchronized (self) {
52  id<SPStateMachineProtocol> previousStateMachine = [self.identifierToStateMachine objectForKey:stateMachineIdentifier];
53  if (previousStateMachine) {
54  if ([stateMachine isMemberOfClass:[previousStateMachine class]]) {
55  return;
56  }
57  [self removeStateMachine:stateMachineIdentifier];
58  }
59  self.identifierToStateMachine[stateMachineIdentifier] = stateMachine;
60  [self.stateMachineToIdentifier setObject:stateMachineIdentifier forKey:stateMachine];
61  [self addToSchemaRegistry:self.eventSchemaToStateMachine
62  schemas:[stateMachine subscribedEventSchemasForTransitions]
63  stateMachine:stateMachine];
64  [self addToSchemaRegistry:self.eventSchemaToEntitiesGenerator
65  schemas:[stateMachine subscribedEventSchemasForEntitiesGeneration]
66  stateMachine:stateMachine];
67  [self addToSchemaRegistry:self.eventSchemaToPayloadUpdater
68  schemas:[stateMachine subscribedEventSchemasForPayloadUpdating]
69  stateMachine:stateMachine];
70  }

73 - (BOOL)removeStateMachine:(NSString *)stateMachineIdentifier {
74  id<SPStateMachineProtocol> stateMachine = self.identifierToStateMachine[stateMachineIdentifier];
75  if (!stateMachine) {
76  return NO;
77  }
78  [self.identifierToStateMachine removeObjectForKey:stateMachineIdentifier];
79  [self.stateMachineToIdentifier removeObjectForKey:stateMachine];
80  [self.trackerState removeStateWithIdentifier:stateMachineIdentifier];
81  [self removeFromSchemaRegistry:self.eventSchemaToStateMachine
82  schemas:[stateMachine subscribedEventSchemasForTransitions]
83  stateMachine:stateMachine];
84  [self removeFromSchemaRegistry:self.eventSchemaToEntitiesGenerator
85  schemas:[stateMachine subscribedEventSchemasForEntitiesGeneration]
86  stateMachine:stateMachine];
87  [self removeFromSchemaRegistry:self.eventSchemaToPayloadUpdater
88  schemas:[stateMachine subscribedEventSchemasForPayloadUpdating]
89  stateMachine:stateMachine];
90  return YES;
91 }

93 - (id<SPTrackerStateSnapshot>)trackerStateForProcessedEvent:(SPEvent *)event {
94  @synchronized (self) {
95  if ([event isKindOfClass:SPSelfDescribingAbstract.class]) {
96  SPSelfDescribingAbstract *sdEvent = (SPSelfDescribingAbstract *)event;
97  NSMutableArray<id<SPStateMachineProtocol>> *stateMachines = self.eventSchemaToStateMachine[sdEvent.schema].mutableCopy ?: [NSMutableArray new];
98  [stateMachines addObjectsFromArray:self.eventSchemaToStateMachine[@"*"]];
99  for (id<SPStateMachineProtocol> stateMachine in stateMachines) {
100  NSString *stateIdentifier = [self.stateMachineToIdentifier objectForKey:stateMachine];
101  SPStateFuture *previousStateFuture = [self.trackerState stateFutureWithIdentifier:stateIdentifier];
102  SPStateFuture *currentStateFuture = [[SPStateFuture alloc] initWithEvent:sdEvent previousState:previousStateFuture stateMachine:stateMachine];
103  [self.trackerState setStateFuture:currentStateFuture identifier:stateIdentifier];
104  // TODO: Remove early state computation.
105  /*
106  The early state-computation causes low performance as it's executed synchronously on
107  the track method thread. Ideally, the state computation should be executed only on
108  entities generation or payload updating (outputs). In that case there are two problems
109  to address:
110  - long chains of StateFuture filling the memory (in case the outputs are not generated)
111  - event object reuse by the user (the event object in the StateFuture could be modified
112  externally)
113  Remove the early state-computation only when these two problems are fixed.
114  */
115  [currentStateFuture state]; // Early state-computation
116  }
117  }
118  return self.trackerState.snapshot;
119  }
120 }

122 - (NSArray<SPSelfDescribingJson *> *)entitiesForProcessedEvent:(id<SPInspectableEvent>)event {
123  @synchronized (self) {
124  NSMutableArray<SPSelfDescribingJson *> *result = [NSMutableArray new];
125  NSMutableArray<id<SPStateMachineProtocol>> *stateMachines = self.eventSchemaToEntitiesGenerator[event.schema].mutableCopy ?: [NSMutableArray new];
126  [stateMachines addObjectsFromArray:self.eventSchemaToEntitiesGenerator[@"*"]];
127  for (id<SPStateMachineProtocol> stateMachine in stateMachines) {
128  NSString *stateIdentifier = [self.stateMachineToIdentifier objectForKey:stateMachine];
129  id<SPState> state = [event.state stateWithIdentifier:stateIdentifier];
130  NSArray<SPSelfDescribingJson *> *entities = [stateMachine entitiesFromEvent:event state:state];
131  if (entities) {
132  [result addObjectsFromArray:entities];
133  }
134  }
135  return result;
136  }

139 - (BOOL)addPayloadValuesToEvent:(id<SPInspectableEvent>)event {
140  @synchronized (self) {
141  int failures = 0;
142  NSMutableArray<id<SPStateMachineProtocol>> *stateMachines = self.eventSchemaToPayloadUpdater[event.schema] ?: [NSMutableArray new];
143  [stateMachines addObjectsFromArray:self.eventSchemaToPayloadUpdater[@"*"]];
144  for (id<SPStateMachineProtocol> stateMachine in stateMachines) {
145  NSString *stateIdentifier = [self.stateMachineToIdentifier objectForKey:stateMachine];
146  id<SPState> state = [event.state stateWithIdentifier:stateIdentifier];
147  NSDictionary<NSString *, NSObject *> *payloadValues = [stateMachine payloadValuesFromEvent:event state:state];
148  if (payloadValues && ![event addPayloadValues:payloadValues]) {
149  failures++;
150  }
151  }
152  return failures == 0;
153  }
154 }
