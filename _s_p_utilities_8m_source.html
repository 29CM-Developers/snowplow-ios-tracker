176 + (NSString *) camelcaseParsedKey:(NSString *)key {
177  NSScanner * scanner = [[NSScanner alloc] initWithString:key];
178  NSMutableArray * words = [[NSMutableArray alloc] init];
179  NSString * scannedWord = [[NSString alloc] init];
180 
181  while (![scanner isAtEnd]) {
182  [scanner scanUpToString:@"-" intoString:&scannedWord];
183  [words addObject:scannedWord];
184  SPLogVerbose(@"scanned word: %@", scannedWord);
185  [scanner scanString:@"-" intoString:nil];
186  }
187 
188  SPLogVerbose(@"%@", words);
189  if ([words count] == 0) {
190  return @"";
191  } else if ([words count] == 1) {
192  return [[NSString alloc] initWithString:[words[0] lowercaseString]];
193  } else {
194  NSMutableString * camelcaseKey = [[NSMutableString alloc] initWithString:[words[0] lowercaseString]];
195  NSRange range;
196  range.length = words.count-1;
197  range.location = 1;
198  for (NSString * word in [words subarrayWithRange:range]) {
199  [camelcaseKey appendString:[word capitalizedString]];
200  }
201  return camelcaseKey;
202  }
203 }
