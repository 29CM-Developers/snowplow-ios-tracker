92 - (instancetype)initWithNamespace:(NSString *)namespace network:(SPNetworkConfiguration *)networkConfiguration configurations:(NSArray<SPConfiguration *> *)configurations {
93  if (self = [super init]) {
94  [self initializeConfigurationUpdates];
95  self.namespace = namespace;
96  self.networkConfigurationUpdate.sourceConfig = networkConfiguration;
97  [self processConfigurations:configurations];
98  if (!self.trackerConfigurationUpdate.sourceConfig) {
99  self.trackerConfigurationUpdate.sourceConfig = [SPTrackerConfiguration new];
100  }
101  [self tracker]; // Build tracker to initialize NotificationCenter receivers
102  }
103  return self;
104 }

124 - (void)processConfigurations:(NSArray<SPConfiguration *> *)configurations {
125  for (SPConfiguration *configuration in configurations) {
126  if ([configuration isKindOfClass:SPNetworkConfiguration.class]) {
127  self.networkConfigurationUpdate.sourceConfig = (SPNetworkConfiguration *)configuration;
128  continue;
129  }
130  if ([configuration isKindOfClass:SPTrackerConfiguration.class]) {
131  self.trackerConfigurationUpdate.sourceConfig = (SPTrackerConfiguration *)configuration;
132  continue;
133  }
134  if ([configuration isKindOfClass:SPSubjectConfiguration.class]) {
135  self.subjectConfigurationUpdate.sourceConfig = (SPSubjectConfiguration *)configuration;
136  continue;
137  }
138  if ([configuration isKindOfClass:SPSessionConfiguration.class]) {
139  self.sessionConfigurationUpdate.sourceConfig = (SPSessionConfiguration *)configuration;
140  continue;
141  }
142  if ([configuration isKindOfClass:SPEmitterConfiguration.class]) {
143  self.emitterConfigurationUpdate.sourceConfig = (SPEmitterConfiguration *)configuration;
144  continue;
145  }
146  if ([configuration isKindOfClass:SPGDPRConfiguration.class]) {
147  self.gdprConfigurationUpdate.sourceConfig = (SPGDPRConfiguration *)configuration;
148  continue;
149  }
150  if ([configuration isKindOfClass:SPGlobalContextsConfiguration.class]) {
151  self.globalContextConfiguration = (SPGlobalContextsConfiguration *)configuration;
152  continue;
153  }
154  }

269 - (SPEmitter *)makeEmitter {
270  SPNetworkConfigurationUpdate *networkConfig = self.networkConfigurationUpdate;
271  SPEmitterConfigurationUpdate *emitterConfig = self.emitterConfigurationUpdate;
272  return [SPEmitter build:^(id<SPEmitterBuilder> builder) {
273  if (networkConfig.networkConnection) {
274  [builder setNetworkConnection:networkConfig.networkConnection];
275  } else {
276  [builder setHttpMethod:networkConfig.method];
277  [builder setProtocol:networkConfig.protocol];
278  [builder setUrlEndpoint:networkConfig.endpoint];
279  }
280  [builder setCustomPostPath:networkConfig.customPostPath];
281  [builder setRequestHeaders:networkConfig.requestHeaders];
282  if (emitterConfig) {
283  [builder setEmitRange:emitterConfig.emitRange];
284  [builder setBufferOption:emitterConfig.bufferOption];
285  [builder setEventStore:emitterConfig.eventStore];
286  [builder setByteLimitPost:emitterConfig.byteLimitPost];
287  [builder setByteLimitGet:emitterConfig.byteLimitGet];
288  [builder setEmitThreadPoolSize:emitterConfig.threadPoolSize];

294 - (SPTracker *)makeTracker {
295  SPEmitter *emitter = self.emitter;
296  SPSubject *subject = self.subject;
297  SPTrackerConfiguration *trackerConfig = self.trackerConfigurationUpdate;
298  SPSessionConfiguration *sessionConfig = self.sessionConfigurationUpdate;
299  SPGlobalContextsConfiguration *gcConfig = self.globalContextConfiguration;
300  SPTracker *tracker = [SPTracker build:^(id<SPTrackerBuilder> builder) {
301  [builder setTrackerNamespace:self.namespace];
302  [builder setEmitter:emitter];
303  [builder setSubject:subject];
304  [builder setAppId:trackerConfig.appId];
305  [builder setTrackerVersionSuffix:trackerConfig.trackerVersionSuffix];
306  [builder setBase64Encoded:trackerConfig.base64Encoding];
307  [builder setLogLevel:trackerConfig.logLevel];
308  [builder setLoggerDelegate:trackerConfig.loggerDelegate];
309  [builder setDevicePlatform:trackerConfig.devicePlatform];
310  [builder setSessionContext:trackerConfig.sessionContext];
311  [builder setApplicationContext:trackerConfig.applicationContext];
312  [builder setScreenContext:trackerConfig.screenContext];
313  [builder setAutotrackScreenViews:trackerConfig.screenViewAutotracking];
314  [builder setLifecycleEvents:trackerConfig.lifecycleAutotracking];
315  [builder setInstallEvent:trackerConfig.installAutotracking];
316  [builder setExceptionEvents:trackerConfig.exceptionAutotracking];
317  [builder setTrackerDiagnostic:trackerConfig.diagnosticAutotracking];
318  if (sessionConfig) {
319  [builder setBackgroundTimeout:sessionConfig.backgroundTimeoutInSeconds];
320  [builder setForegroundTimeout:sessionConfig.foregroundTimeoutInSeconds];
321  }
322  if (gcConfig) {
323  [builder setGlobalContextGenerators:gcConfig.contextGenerators];
324  }
325  SPGDPRConfigurationUpdate *gdprConfig = self.gdprConfigurationUpdate;
326  if (gdprConfig.sourceConfig) {
327  [builder setGdprContextWithBasis:gdprConfig.basisForProcessing documentId:gdprConfig.documentId documentVersion:gdprConfig.documentVersion documentDescription:gdprConfig.documentDescription];
328  }
329  }];
330  if (self.trackerConfigurationUpdate.isPaused) {
331  [tracker pauseEventTracking];
332  }
333  if (self.sessionConfigurationUpdate.isPaused) {
334  [tracker.session stopChecker];
335  }
336  return tracker;
337 }
