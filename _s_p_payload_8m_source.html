86 - (void) addJsonToPayload:(NSData *)json
87  base64Encoded:(Boolean)encode
88  typeWhenEncoded:(NSString *)typeEncoded
89  typeWhenNotEncoded:(NSString *)typeNotEncoded {
90  NSDictionary *object = [SPJSONSerialization deserializeData:json];
91  if (!object) {
92  return;
93  }
94  if (encode) {
95  NSString *encodedString = [json base64EncodedStringWithOptions:0];
96 
97  // We need URL safe with no padding. Since there is no built-in way to do this, we transform
98  // the encoded payload to make it URL safe by replacing chars that are different in the URL-safe
99  // alphabet. Namely, 62 is - instead of +, and 63 _ instead of /.
100  // See: https://tools.ietf.org/html/rfc4648#section-5
101  encodedString = [[encodedString stringByReplacingOccurrencesOfString:@"/" withString:@"_"]
102  stringByReplacingOccurrencesOfString:@"+" withString:@"-"];
103 
104  // There is also no padding since the length is implicitly known.
105  encodedString = [encodedString stringByTrimmingCharactersInSet:[NSCharacterSet characterSetWithCharactersInString:@"="]];
106 
107  [self addValueToPayload:encodedString forKey:typeEncoded];
108  } else {
109  [self addValueToPayload:[[NSString alloc] initWithData:json encoding:NSUTF8StringEncoding] forKey:typeNotEncoded];
110  }
111 }
