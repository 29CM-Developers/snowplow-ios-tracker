78 + (NSString *)stringFromNotificationTrigger:(nullable UNNotificationTrigger *)trigger API_AVAILABLE(ios(10.0)) {
79  NSMutableString * triggerType = [[NSMutableString alloc] initWithString:@"UNKNOWN"];
80  NSString * triggerClass = NSStringFromClass([trigger class]);

82  [triggerType setString:@"TIME_INTERVAL"];
