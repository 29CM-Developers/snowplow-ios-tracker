75 - (instancetype)initWithForegroundTimeout:(NSInteger)foregroundTimeout andBackgroundTimeout:(NSInteger)backgroundTimeout andTracker:(SPTracker *)tracker {
76  if (self = [super init]) {
77  _foregroundTimeout = foregroundTimeout * 1000;
78  _backgroundTimeout = backgroundTimeout * 1000;
79  _inBackground = NO;
80  _isNewSession = YES;
81  _sessionStorage = @"LOCAL_STORAGE";
82  self.tracker = tracker;
83  self.dataPersistence = [SPDataPersistence dataPersistenceForNamespace:tracker.trackerNamespace];
84  NSDictionary *storedSessionDict = self.dataPersistence.session;
85 
86  if (storedSessionDict) {
87  _userId = [storedSessionDict valueForKey:kSPSessionUserId] ?: [SPUtilities getUUIDString];
88  _currentSessionId = [storedSessionDict valueForKey:kSPSessionId];
89  _sessionIndex = [[storedSessionDict valueForKey:kSPSessionIndex] intValue];
90  } else {
91  _userId = [SPUtilities getUUIDString];
92  _currentSessionId = nil;
93  _sessionIndex = 0;
94  }
95 
96  // Get or Set the Session UserID
97  NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
98  NSString *storedUserId = [userDefaults stringForKey:kSPInstallationUserId];
99  if (storedUserId) {
100  _userId = storedUserId;
101  } else if (_userId) {
102  [userDefaults setObject:_userId forKey:kSPInstallationUserId];
103  }
104 
105  // Start session check
106  self.lastSessionCheck = [SPUtilities getTimestamp];
107  [self startChecker];
108 
109  // Trigger notification for view changes
110  #if SNOWPLOW_TARGET_IOS || SNOWPLOW_TARGET_TV

214 - (BOOL)shouldUpdateSession {
215  if (_isNewSession) {
216  return YES;
217  }
218  long long lastAccess = self.lastSessionCheck.longLongValue;
219  long long now = [SPUtilities getTimestamp].longLongValue;
220  NSInteger timeout = _inBackground ? _backgroundTimeout : _foregroundTimeout;
221  return now < lastAccess || now - lastAccess > timeout;
222 }

224 - (void)updateSessionWithEventId:(NSString *)eventId {
225  _isNewSession = NO;
226  _firstEventId = eventId;
227  _previousSessionId = _currentSessionId;
228  _currentSessionId = [SPUtilities getUUIDString];
229  _sessionIndex++;
230 
231  // Update session dictionary used as context for events
232  NSMutableDictionary *newSessionDict = [NSMutableDictionary new];
233  if (_firstEventId) {
234  [newSessionDict setObject:_firstEventId forKey:kSPSessionFirstEventId];
235  }
236  [newSessionDict setObject:_currentSessionId forKey:kSPSessionId];
237  [newSessionDict setObject:(_previousSessionId != nil ? _previousSessionId : [NSNull null]) forKey:kSPSessionPreviousId];
238  [newSessionDict setObject:[NSNumber numberWithInt:(int)_sessionIndex] forKey:kSPSessionIndex];
239  [newSessionDict setObject:_sessionStorage forKey:kSPSessionStorage];
240  _sessionDict = [newSessionDict copy];
241 
242  [self writeSessionToFile];
243 }
