82 - (void) setupUrlEndpoint {
83  NSString * urlPrefix = _protocol == SPHttp ? @"http://" : @"https://";
84  NSString * urlSuffix = _httpMethod == SPRequestGet ? kSPEndpointGet : kSPEndpointPost;
85  if (_customPostPath && _httpMethod == SPRequestPost) {
86  urlSuffix = _customPostPath;
87  }
88  _urlEndpoint = [NSURL URLWithString:[NSString stringWithFormat:@"%@%@%@", urlPrefix, _url, urlSuffix]];
89 
90  if (_urlEndpoint && _urlEndpoint.scheme && _urlEndpoint.host) {
91  SnowplowDLog(@"SPLog: Emitter URL created successfully '%@'", _urlEndpoint);
92  NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
93  [userDefaults setObject:_url forKey:kSPErrorTrackerUrl];
94  [userDefaults setObject:urlSuffix forKey:kSPErrorTrackerProtocol];
95  [userDefaults setObject:urlPrefix forKey:kSPErrorTrackerMethod];
96  } else {
97  SnowplowDLog(@"SPLog: Invalid emitter URL: '%@'", _urlEndpoint);
98  NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
99  [userDefaults setObject:@"acme.com" forKey:kSPErrorTrackerUrl];
100  [userDefaults setObject:kSPEndpointPost forKey:kSPErrorTrackerProtocol];
101  [userDefaults setObject:@"http://" forKey:kSPErrorTrackerMethod];
102  }
103 }

190 - (void) sendEvents {
191  SnowplowDLog(@"SPLog: Sending events...");
192 
193  if ([self getDbCount] == 0) {
194  SnowplowDLog(@"SPLog: Database empty. Returning..");
195  _isSending = NO;
196  return;
197  }
198 
199  NSArray *listValues = [[NSArray alloc] initWithArray:[_db getAllEventsLimited:_emitRange]];
200  NSMutableArray *sendResults = [[NSMutableArray alloc] init];
201 
202  if (_httpMethod == SPRequestPost) {
203  NSMutableArray *eventArray = [[NSMutableArray alloc] init];
204  NSMutableArray *indexArray = [[NSMutableArray alloc] init];
205  NSInteger totalByteSize = 0;
206 
207  for (int i = 0; i < listValues.count; i ++) {
208 
209  // Get the event payload
210  NSMutableDictionary *eventPayload = [[listValues[i] objectForKey:@"eventData"] mutableCopy];
211 
212  // Convert to NSData and check the byte size
213  NSData *data = [NSJSONSerialization dataWithJSONObject:eventPayload options:0 error:nil];
214  NSInteger payloadByteSize = [SPUtilities getByteSizeWithString:[[NSString alloc] initWithData:data encoding:NSUTF8StringEncoding]];
215  payloadByteSize += POST_STM_BYTES;
216 
217  if ((payloadByteSize + POST_WRAPPER_BYTES) > _byteLimitPost) {
218  // Single event exceeds the byte limit so must be sent individually.
219  NSMutableArray *singleEventArray = [[NSMutableArray alloc] init];
220  NSMutableArray *singleIndexArray = [[NSMutableArray alloc] init];
221 
222  // Build and Send the event!
223  [singleEventArray addObject:eventPayload];
224  [singleIndexArray addObject:[listValues[i] objectForKey:@"ID"]];
225 
226  // Add the STM to the event
227  [self addStmToEventPayloadsWithArray:singleEventArray];
228 
229  SPSelfDescribingJson *payload = [[SPSelfDescribingJson alloc] initWithSchema:kSPPayloadDataSchema
230  andData:singleEventArray];
231  [self sendEventWithRequest:[self getRequestPostWithData:payload] andIndex:singleIndexArray andResultArray:sendResults andOversize:YES];
232  } else if ((totalByteSize + payloadByteSize + POST_WRAPPER_BYTES + (eventArray.count - 1)) > _byteLimitPost) {
233  // Add the STM to each event
234  [self addStmToEventPayloadsWithArray:eventArray];
235 
236  // Adding this event to the accumulated array would exceed the limit.
237  SPSelfDescribingJson *payload = [[SPSelfDescribingJson alloc] initWithSchema:kSPPayloadDataSchema
238  andData:eventArray];
239  [self sendEventWithRequest:[self getRequestPostWithData:payload] andIndex:indexArray andResultArray:sendResults andOversize:NO];
240 
241  // Reset collections and STM
242  eventArray = [[NSMutableArray alloc] init];
243  indexArray = [[NSMutableArray alloc] init];
244 
245  // Add event to collections
246  [eventArray addObject:eventPayload];
247  [indexArray addObject:[listValues[i] objectForKey:@"ID"]];
248 
249  // Update byte count
250  totalByteSize = payloadByteSize;
251  } else {
252  // Add event to collections
253  [eventArray addObject:eventPayload];
254  [indexArray addObject:[listValues[i] objectForKey:@"ID"]];
255 
256  // Update byte count
257  totalByteSize += payloadByteSize;
258  }
259  }
260 
261  // If we have not sent all of the events...
262  if (eventArray.count > 0) {
263  // Add the STM to each event
264  [self addStmToEventPayloadsWithArray:eventArray];
265 
266  // Send the event!
267  SPSelfDescribingJson *payload = [[SPSelfDescribingJson alloc] initWithSchema:kSPPayloadDataSchema
268  andData:eventArray];
269  [self sendEventWithRequest:[self getRequestPostWithData:payload] andIndex:indexArray andResultArray:sendResults andOversize:NO];
270  }
271  } else {
272  for (NSDictionary * eventWithMetaData in listValues) {
273  NSArray *indexArray = [NSArray arrayWithObject:[eventWithMetaData objectForKey:@"ID"]];
274  NSMutableDictionary *eventPayload = [[eventWithMetaData objectForKey:@"eventData"] mutableCopy];
275  [eventPayload setValue:[NSString stringWithFormat:@"%lld", [[SPUtilities getTimestamp] longLongValue]] forKey:kSPSentTimestamp];
276 
277  // Make GET URL to send
278  NSString *url = [NSString stringWithFormat:@"%@?%@", [_urlEndpoint absoluteString], [SPUtilities urlEncodeDictionary:eventPayload]];
279  BOOL oversize = ([SPUtilities getByteSizeWithString:url] > _byteLimitGet);
280 
281  // Send the request
282  [self sendEventWithRequest:[self getRequestGetWithString:url] andIndex:indexArray andResultArray:sendResults andOversize:oversize];
283  }
284  }
285 
286  [_dataOperationQueue waitUntilAllOperationsAreFinished];
287 
288  NSInteger success = 0;
289  NSInteger failure = 0;
290 
291  for (int i = 0; i < sendResults.count; i++) {
292  SPRequestResponse * result = [sendResults objectAtIndex:i];
293  NSArray * resultIndexArray = [result getIndexArray];
294 
295  if ([result getSuccess]) {
296  success += resultIndexArray.count;
297  [self processSuccessesWithResults:resultIndexArray];
298  } else {
299  failure += resultIndexArray.count;
300  }
301  }
302 
303  [_dataOperationQueue waitUntilAllOperationsAreFinished];
304 
305  SnowplowDLog(@"SPLog: Emitter Success Count: %@", [@(success) stringValue]);
306  SnowplowDLog(@"SPLog: Emitter Failure Count: %@", [@(failure) stringValue]);
307 
308  if (_callback != nil) {
309  if (failure == 0) {
310  [_callback onSuccessWithCount:success];
311  } else {
312  [_callback onFailureWithCount:failure successCount:success];
313  }
314  }
315 
316  [sendResults removeAllObjects];
317 
318  if (success == 0 && failure > 0) {
319  SnowplowDLog(@"SPLog: Ending emitter run as all requests failed...");
320  [NSThread sleepForTimeInterval:5];
321  _isSending = NO;
322  return;
323  } else {
324  [self sendEvents];
325  }
326 }

328 - (void) sendEventWithRequest:(NSMutableURLRequest *)request andIndex:(NSArray *)indexArray andResultArray:(NSMutableArray *)results andOversize:(BOOL)oversize {
329  [_dataOperationQueue addOperationWithBlock:^{
330  //source: https://forums.developer.apple.com/thread/11519
331  __block NSHTTPURLResponse *httpResponse = nil;
332  __block NSError *connectionError = nil;
333  dispatch_semaphore_t sem;
334 
335  sem = dispatch_semaphore_create(0);
336 
337  [[[NSURLSession sharedSession] dataTaskWithRequest:request
338  completionHandler:^(NSData *data, NSURLResponse *urlResponse, NSError *error) {
339 
340  connectionError = error;
341  httpResponse = (NSHTTPURLResponse*)urlResponse;
342  dispatch_semaphore_signal(sem);
343  }] resume];
344 
345  dispatch_semaphore_wait(sem, DISPATCH_TIME_FOREVER);
346 
347  @synchronized (results) {
348  if (oversize) {
349  [results addObject:[[SPRequestResponse alloc] initWithBool:true withIndex:indexArray]];
350  } else if ([httpResponse statusCode] >= 200 && [httpResponse statusCode] < 300) {
351  [results addObject:[[SPRequestResponse alloc] initWithBool:true withIndex:indexArray]];
352  } else {
353  NSLog(@"SPLog: Error: %@", connectionError);
354  [results addObject:[[SPRequestResponse alloc] initWithBool:false withIndex:indexArray]];
355  }
356  }
357  }];
358 }
