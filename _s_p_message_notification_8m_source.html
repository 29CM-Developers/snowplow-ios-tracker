90 - (NSDictionary<NSString *, NSObject *> *)payload {
91  NSMutableDictionary *payload = [NSMutableDictionary dictionary];
92  [payload setValue:_action forKey:kSPMessageNotificationParamAction];
93  if (_attachments.count > 0) {
94  NSMutableArray<NSDictionary<NSString *, NSObject *> *> *dictAttachments = [NSMutableArray array];
95  for (SPMessageNotificationAttachment *attachment in _attachments) {
96  [dictAttachments addObject:attachment.data];
97  }
98  [payload setValue:dictAttachments forKey:kSPMessageNotificationParamMessageNotificationAttachments];
99  }
100  [payload setValue:_body forKey:kSPMessageNotificationParamBody];
101  if (_bodyLocArgs.count > 0) {
102  [payload setValue:_bodyLocArgs forKey:kSPMessageNotificationParamBodyLocArgs];
103  }
104  [payload setValue:_bodyLocKey forKey:kSPMessageNotificationParamBodyLocKey];
105  [payload setValue:_category forKey:kSPMessageNotificationParamCategory];
106  if (_contentAvailable) {
107  NSNumber *contentAvailableBoolTyped = @(_contentAvailable.boolValue);
108  [payload setValue:contentAvailableBoolTyped forKey:kSPMessageNotificationParamContentAvailable];
109  }
110  [payload setValue:_group forKey:kSPMessageNotificationParamGroup];
111  [payload setValue:_icon forKey:kSPMessageNotificationParamIcon];
112  [payload setValue:_notificationCount forKey:kSPMessageNotificationParamNotificationCount];
113  [payload setValue:_notificationTimestamp forKey:kSPMessageNotificationParamNotificationTimestamp];
114  [payload setValue:_sound forKey:kSPMessageNotificationParamSound];
115  [payload setValue:_subtitle forKey:kSPMessageNotificationParamSubtitle];
116  [payload setValue:_tag forKey:kSPMessageNotificationParamTag];
117  [payload setValue:_threadIdentifier forKey:kSPMessageNotificationParamThreadIdentifier];
118  [payload setValue:_title forKey:kSPMessageNotificationParamTitle];
119  if (_titleLocArgs.count > 0) {

123  [payload setValue:SPMessageNotificationTriggerString(_trigger) forKey:kSPMessageNotificationParamTrigger];
124  return payload;

129 + (SPMessageNotification *)messageNotificationWithUserInfo:(NSDictionary *)userInfo defaultTitle:(NSString *)defaultTitle defaultBody:(NSString *)defaultBody {
130  NSDictionary *aps = [userInfo sp_dictionaryForKey:@"aps" defaultValue:nil];
131  if (!aps) {
132  return nil;
133  }
134  NSDictionary *alert = [aps sp_dictionaryForKey:@"alert" defaultValue:nil];
135  if (!alert) {
136  return nil;
137  }
138  // alert fields
139  NSString *title = [alert sp_stringForKey:@"title" defaultValue:defaultTitle];
140  NSString *body = [alert sp_stringForKey:@"body" defaultValue:defaultBody];
141  if (!title || !body) {
142  return nil;
143  }
144  NSString *subtitle = [alert sp_stringForKey:@"subtitle" defaultValue:nil];
145  NSString *launchImage = [alert sp_stringForKey:@"launch-image" defaultValue:nil];
146  NSString *titleLocKey = [alert sp_stringForKey:@"title-loc-key" defaultValue:nil];
147  NSArray *titleLocArgs = [alert sp_arrayForKey:@"title-loc-args" itemClass:NSString.class defaultValue:nil];
148  NSString *locKey = [alert sp_stringForKey:@"loc-key" defaultValue:nil];
149  NSArray *locArgs = [alert sp_arrayForKey:@"loc-args" itemClass:NSString.class defaultValue:nil];
150  // aps fields
151  NSNumber *badge = [aps sp_numberForKey:@"badge" defaultValue:nil];
152  NSString *sound = [aps sp_stringForKey:@"sound" defaultValue:nil];
153  NSNumber *contentAvailable = [aps sp_numberForKey:@"content-available" defaultValue:nil];
154  NSString *category = [aps sp_stringForKey:@"category" defaultValue:nil];
155  NSString *threadId = [aps sp_stringForKey:@"thread-id" defaultValue:nil];
156  SPMessageNotification *event = [[SPMessageNotification alloc] initWithTitle:title body:body trigger:SPMessageNotificationTriggerPush];
157  event.subtitle = subtitle;
158  event.icon = launchImage;
159  event.titleLocKey = titleLocKey;
160  event.titleLocArgs = titleLocArgs;
161  event.bodyLocKey = locKey;
162  event.bodyLocArgs = locArgs;
163  event.notificationCount = badge;
164  event.sound = sound;
165  event.contentAvailable = contentAvailable;
166  event.category = category;
167  event.threadIdentifier = threadId;
168  return event;
169 }
