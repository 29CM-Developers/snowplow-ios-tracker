70 void uncaughtExceptionHandler(NSException *exception) {
71  NSArray* symbols = [exception callStackSymbols];
72  NSString * stackTrace = [NSString stringWithFormat:@"Stacktrace:\n%@", symbols];
73  NSString * message = [exception reason];
74  dispatch_sync(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
75  // Load values
76  NSUserDefaults * userDefaults = [NSUserDefaults standardUserDefaults];
77  NSString * url = [userDefaults objectForKey:kSPErrorTrackerUrl];
78  NSString * protocolString = [userDefaults objectForKey:kSPErrorTrackerProtocol];
79  NSString * methodString = [userDefaults objectForKey:kSPErrorTrackerMethod];
80  SPProtocol protocol = SPHttps;
81  if (protocolString && [protocolString isEqual:@"http://"]) {
82  protocol = SPHttp;
83  }
84  SPRequestOptions method = SPRequestPost;
85  if (methodString && [methodString isEqual:@"/i"]) {
86  method = SPRequestGet;
87  }
88  // Send notification to tracker
89  SPEmitter * emitter = [SPEmitter build:^(id<SPEmitterBuilder> builder) {
90  [builder setUrlEndpoint:url];
91  [builder setProtocol:protocol];
92  [builder setHttpMethod:method];
93  }];
94  SPTracker * tracker = [SPTracker build:^(id<SPTrackerBuilder> builder) {
95  [builder setEmitter:emitter];
96  }];
97 
98  if (message == nil || [message length] == 0) {
99  return;
100  }
101  SNOWError * error = [SNOWError build:^(id<SPErrorBuilder> builder) {
102  [builder setMessage:message];
103  if (stackTrace != nil && [stackTrace length] > 0) {
104  [builder setStackTrace:stackTrace];
105  }
106  }];
107  [tracker trackErrorEvent:error];
108  [NSThread sleepForTimeInterval:2.0f];
109  });
110 }

195 - (void) checkInstall {
196  SPInstallTracker * installTracker = [[SPInstallTracker alloc] init];
197  NSNumber * previousTimestamp = [installTracker getPreviousInstallTimestamp];
198  if (_installEvent) {
199  if (installTracker.isNewInstall) {
200  SPSelfDescribingJson * installEvent = [[SPSelfDescribingJson alloc] initWithSchema:kSPApplicationInstallSchema andData:@{}];
201  SPUnstructured * event = [SPUnstructured build:^(id<SPUnstructuredBuilder> builder) {
202  [builder setEventData:installEvent];
203  }];
204  [self trackUnstructuredEvent:event];
205  if (previousTimestamp) {
206  [installTracker clearPreviousInstallTimestamp];
207  }
208  } else if (previousTimestamp) {
209  SPSelfDescribingJson * installEvent = [[SPSelfDescribingJson alloc] initWithSchema:kSPApplicationInstallSchema andData:@{}];
210  SPUnstructured * event = [SPUnstructured build:^(id<SPUnstructuredBuilder> builder) {
211  [builder setEventData:installEvent];
212  [builder setTimestamp:previousTimestamp];
213  }];
214  [self trackUnstructuredEvent:event];
215  [installTracker clearPreviousInstallTimestamp];
216  }
217  }

543 - (SPPayload *)payloadWithEvent:(SPTrackerEvent *)event {
544  SPPayload *payload = [SPPayload new];
545 
546  [self addBasicPropertiesToPayload:payload event:event];
547  if (event.isPrimitive) {
548  [self addPrimitivePropertiesToPayload:payload event:event];

550  [self addSelfDescribingPropertiesToPayload:payload event:event];

553  [self addBasicContextsToContexts:contexts event:event];

592 - (void)addBasicContextsToContexts:(NSMutableArray<SPSelfDescribingJson *> *)contexts eventId:(NSString *)eventId {
593  if (_subject != nil) {
594  NSDictionary * platformDict = [[_subject getPlatformDict] getAsDictionary];
595  if (platformDict != nil) {
596  [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:_platformContextSchema andData:platformDict]];
597  }
598  NSDictionary * geoLocationDict = [_subject getGeoLocationDict];
599  if (geoLocationDict != nil) {
600  [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPGeoContextSchema andData:geoLocationDict]];
601  }
602  }
603 
604  if (_applicationContext) {
605  SPSelfDescribingJson * contextJson = [SPUtilities getApplicationContext];
606  if (contextJson != nil) {
607  [contexts addObject:contextJson];
608  }
609  }
610 
611  // Add session if active
612  if (_session != nil) {
613  NSDictionary * sessionDict = [_session getSessionDictWithEventId:eventId];
614  if (sessionDict != nil) {
615  [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPSessionContextSchema andData:sessionDict]];
616  }
617  }
618 
619  // Add screen context
620  if (_screenContext && _currentScreenState) {
621  SPSelfDescribingJson * contextJson = [SPUtilities getScreenContextWithScreenState:_currentScreenState];
622  if (contextJson != nil) {
623  [contexts addObject:contextJson];
624  }
625  }
626 
627  // Add GDPR context
628  if (self.gdpr) {
629  SPSelfDescribingJson *gdprContext = self.gdpr.context;
630  if (gdprContext) {
631  [contexts addObject:gdprContext];
632  }
633  }

642 - (void)wrapContexts:(NSArray<SPSelfDescribingJson *> *)contexts toPayload:(SPPayload *)payload {
643  if (contexts.count == 0) {
644  return;
645  }
646  NSMutableArray<NSDictionary *> *data = [NSMutableArray new];
